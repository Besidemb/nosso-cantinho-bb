    <script>
        // Atualiza moedas
        function alterarMoedas(perfil, valor) {
            const span = document.getElementById(`moedas-${perfil}`);
            let total = parseInt(span.textContent);
            total = Math.max(0, total + valor);
            span.textContent = total;
            salvarDados();
        }

        // Salva notas e moedas no navegador
        function salvarDados() {
            const dados = {
                mb: {
                    notas: document.getElementById('notas-mb').innerText,
                    moedas: document.getElementById('moedas-mb').textContent
                },
                nath: {
                    notas: document.getElementById('notas-nath').innerText,
                    moedas: document.getElementById('moedas-nath').textContent
                }
            };
            localStorage.setItem('nossoCantinho', JSON.stringify(dados));
        }

        // Carrega dados salvos
        window.onload = function() {
            const salvos = localStorage.getItem('nossoCantinho');
            if (salvos) {
                const dados = JSON.parse(salvos);
                document.getElementById('notas-mb').innerText = dados.mb.notas;
                document.getElementById('moedas-mb').textContent = dados.mb.moedas;
                document.getElementById('notas-nath').innerText = dados.nath.notas;
                document.getElementById('moedas-nath').textContent = dados.nath.moedas;
            }
        };

        // Salva automaticamente quando sai da pÃ¡gina
        window.addEventListener('beforeunload', salvarDados);
    </script>
